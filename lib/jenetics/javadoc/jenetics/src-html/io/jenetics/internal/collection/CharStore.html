20 package io.jenetics.internal.collection;
21 
22 import static java.util.Objects.requireNonNull;
23 
24 import java.io.Serializable;
25 import java.util.Arrays;
26 import java.util.Comparator;
27 
28 /**
29  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
30  * @since 3.0
31  * @version 3.4
32  */
33 public final class CharStore implements Array.Store<Character>, Serializable {
34     private static final long serialVersionUID = 1L;
35 
36     public final char[] array;
37 
38     private CharStore(final char[] chars) {
39         array = requireNonNull(chars);
40     }
41 
42     public CharStore(final int length) {
43         this(new char[length]);
44     }
45 
46     @Override
47     public Character get(final int index) {
48         return array[index];
49     }
50 
51     @Override
52     public void sort(
53         final int from,
54         final int until,
55         final Comparator<? super Character> comparator
56     ) {
57         if (comparator == null) {
58             Arrays.sort(array, from, until);
59         } else {
60             final Character[] chars = new Character[array.length];
61             for (int i = 0; i < array.length; ++i) {
62                 chars[i] = array[i];
63             }
64             Arrays.sort(chars, from, until, comparator);
65             for (int i = 0; i < array.length; ++i) {
66                 array[i] = chars[i];
67             }
68         }
69     }
70 
71     @Override
72     public void set(final int index, final Character value) {
73         array[index] = value;
74     }
75 
76     @Override
77     public CharStore copy(final int from, final int until) {
78         final char[] array = new char[until - from];
79         System.arraycopy(this.array, from, array, 0, until - from);
80         return new CharStore(array);
81     }
82 
83     @Override
84     public CharStore newInstance(final int length) {
85         return new CharStore(length);
86     }
87 
88     @Override
89     public int length() {
90         return array.length;
91     }
92 
93     public static CharStore of(final char[] chars) {
94         return new CharStore(chars);
95     }
96 
97 }
