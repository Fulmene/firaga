050 /**
051  * Represents an function testing measurement environment.
052  *
053  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
054  * @version 4.0
055  * @since 3.4
056  */
057 public final class TrialMeter<T> {
058 
059     private final String _name;
060     private final String _description;
061     private final Env _env;
062 
063     private final Params<T> _params;
064     private final DataSet _dataSet;
065 
066     private TrialMeter(
067         final String name,
068         final String description,
069         final Env env,
070         final Params<T> params,
071         final DataSet dataSet
072     ) {
073         _name = requireNonNull(name);
074         _description = description;
075         _env = requireNonNull(env);
076         _params = requireNonNull(params);
077         _dataSet = requireNonNull(dataSet);
078     }
079 
080     public String getName() {
081         return _name;
082     }
083 
084     /**
085      * Return the optional description string.
086      *
087      * @return the optional description string
088      */
089     public Optional<String> getDescription() {
090         return Optional.ofNullable(_description);
091     }
092 
093     /**
094      * The trial meter environment information.
095      *
096      * @return the trial meter environment information
097      */
098     public Env getEnv() {
099         return _env;
100     }
101 
102     /**
103      * Return the testing parameters.
104      *
105      * @return the testing parameters
106      */
107     public Params<T> getParams() {
108         return _params;
109     }
110 
111     /**
112      * Return the current trail {@link DataSet}.
113      *
114      * @return the current trail data set
115      */
116     public DataSet getDataSet() {
117         return _dataSet;
118     }
119 
120     /**
121      * Return the test data with the given name
122      *
123      * @param name the data name
124      * @return the test {@link Data} with the given name
125      * @throws NullPointerException if the given {@code name} is {@code null}
126      */
127     public Data getData(final String name) {
128         return _dataSet.get(name);
129     }
130 
131     /**
132      * Return the number of test data results.
133      *
134      * @return the number of test data results.
135      */
136     public int dataSize() {
137         return _dataSet.dataSize();
138     }
139 
140     /**
141      * Calculates the test values for all parameters. The length of the
142      * resulting {@code double[]} array must be {@link #dataSize()}.
143      *
144      * @param function the test function
145      */
146     public void sample(final Function<T, double[]> function) {
147         _params.values()
148             .subSeq(_dataSet.nextParamIndex())
149             .forEach(p -> _dataSet.add(function.apply(p)));
150     }
151 
152     @Override
153     public String toString() {
154         return format(
155             "TrialMeter[sample=%d, param=%d]",
156             dataSize(), _dataSet.nextParamIndex()
157         );
158     }
159 
160     /**
161      * Writes the current {@code TrialMeter} object (the calculated samples +
162      * the parameters) to the given output stream.
163      *
164      * @param out the output stream where to write the trial meter
165      * @param writer the writer of the parameter type
166      * @throws UncheckedIOException if the marshalling fails
167      */
168     public void write(final OutputStream out, final Writer<? super T> writer) {
169         try (AutoCloseableXMLStreamWriter xml = XML.writer(out, "    ")) {
170             TrialMeter.<T>writer(writer).write(xml, this);
171         } catch (XMLStreamException e) {
172             throw new UncheckedIOException(new IOException(e));
173         }
174     }
175 
176     /**
177      * Writes the current {@code TrialMeter} object (the calculated samples +
178      * the parameters) to the given path.
179      *
180      * @param path the output path
181      * @param writer the writer of the parameter type
182      * @throws UncheckedIOException if the marshalling fails
183      */
184     public void write(final Path path, final Writer<? super T> writer) {
185         try {
186             final File tempFile = createTempFile("__trial_meter__", ".xml");
187             try {
188                 try (OutputStream out = new FileOutputStream(tempFile)) {
189                     write(out, writer);
190                 }
191 
192                 move(tempFile.toPath(), path, REPLACE_EXISTING);
193             } finally {
194                 deleteIfExists(tempFile.toPath());
195             }
196         } catch (IOException e) {
197             throw new UncheckedIOException(e);
198         }
199     }

201     /**
202      * Return a new trial measure environment.
203      *
204      * @param name the trial meter name
205      * @param description the trial meter description, maybe {@code null}
206      * @param params the parameters which are tested by this trial meter
207      * @param dataSetNames the names of the calculated data sets
208      * @param <T> the parameter type
209      * @return a new trial measure environment
210      */
211     public static <T> TrialMeter<T> of(
212         final String name,
213         final String description,
214         final Params<T> params,
215         final String... dataSetNames
216     ) {
217         return new TrialMeter<T>(
218             name,
219             description,
220             Env.of(),
221             params,
222             DataSet.of(params.size(), dataSetNames)
223         );
224     }

227      * Return a new trial measure environment.
228      *
229      * @param name the trial meter name

231      * @param env the environment information
232      * @param params the parameters which are tested by this trial meter
233      * @param dataSetNames the names of the calculated data sets
234      * @param <T> the parameter type
235      * @return a new trial measure environment
236      */
237     public static <T> TrialMeter<T> of(

253     /**
254      * Read existing {@code TrialMeter} (intermediate) results from the given
255      * input stream.
256      *
257      * @param in the {@link InputStream} to read from
258      * @param reader the writer of the parameter type
259      * @param <T> the parameter type
260      * @throws UncheckedIOException if reading the {@code TrialMeter} fails
261      * @return the {@code TrialMeter} object read from the input stream
262      */
263     @SuppressWarnings("unchecked")
264     public static <T> TrialMeter<T> read(
265         final InputStream in,
266         final Reader<? extends T> reader
267     ) {
268         try (AutoCloseableXMLStreamReader xml = XML.reader(in)) {
269             xml.next();
270             return TrialMeter.<T>reader(reader).read(xml);
271         } catch (XMLStreamException e) {
272             throw new UncheckedIOException(new IOException(e));
273         }
274     }
275 
276     /**
277      * Read existing {@code TrialMeter} (intermediate) results from the given
278      * path.
279      *
280      * @param path the path the {@code TrialMeter} is read
281      * @param <T> the parameter type
282      * @param reader the writer of the parameter type
283      * @throws UncheckedIOException if reading the {@code TrialMeter} fails
284      * @return the {@code TrialMeter} object read from the input stream
285      */
286     public static <T> TrialMeter<T> read(
287         final Path path,
288         final Reader<? extends T> reader
289     ) {
290         try (InputStream in = new FileInputStream(path.toFile())) {
291             return read(in, reader);
292         } catch (IOException e) {
293             throw new UncheckedIOException(e);
294         }
295     }

298     /* *************************************************************************
299      *  XML reader/writer
300      * ************************************************************************/
301 
302     public static <T> Writer<TrialMeter<T>> writer(final Writer<? super T> writer) {
303         return elem(
304             "measurement",
305             attr("name").map(TrialMeter::getName),
306             attr("description").map(tm -> tm.getDescription().orElse("")),
307             Env.WRITER.map(TrialMeter::getEnv),
308             Params.<T>writer(writer).map(TrialMeter::getParams),
309             DataSet.WRITER.map(TrialMeter::getDataSet)
310         );
311     }
312 
313     @SuppressWarnings({"unchecked", "rawtypes"})
314     public static <T> Reader<TrialMeter<T>> reader(final Reader<? extends T> reader) {
315         return Reader.elem(
316             (Object[] v) -> new TrialMeter(
317                 (String)v[0],
318                 (String)v[1],
319                 (Env)v[2],
320                 (Params)v[3],
321                 (DataSet)v[4]
322             ),
323             "measurement",
324             Reader.attr("name"),
325             Reader.attr("description"),
326             Env.READER,
327             Params.reader(reader),
328             DataSet.READER
329         );
330     }
