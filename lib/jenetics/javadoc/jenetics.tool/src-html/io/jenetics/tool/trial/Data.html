037 /**
038  * This class collects a list of {@link Sample} result objects into on
039  * {@code Data} object.
040  *
041  * @see Sample
042  *
043  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
044  * @version 4.0
045  * @since 3.4
046  */
047 public final class Data implements Serializable {
048 
049     private static final long serialVersionUID = 1L;
050 
051     private final String _name;
052     private final List<Sample> _samples = new ArrayList<>();
053 
054     private Data(final String name, final List<Sample> samples) {
055         if (samples.isEmpty()) {
056             throw new IllegalArgumentException("Sample list must not be empty.");
057         }
058         if (!samples.stream().allMatch(s -> samples.get(0).size() == s.size())) {
059             throw new IllegalArgumentException(
060                 "All sample object must have the same size."
061             );
062         }
063 
064         _name = requireNonNull(name);
065         _samples.addAll(samples);
066     }
067 
068     /**
069      * Return the name of the sample {@code Data} collections.
070      *
071      * @return the name of the sample {@code Data} collections
072      */
073     public String getName() {
074         return _name;
075     }
076 
077     /**
078      * The number of {@link Sample} objects this {@code Data} class contains.
079      *
080      * @return the size of the data object
081      */
082     public int dataSize() {
083         return _samples.size();
084     }
085 
086     /**
087      * The number of values of an {@link Sample} object.
088      *
089      * @see Sample#size()
090      *
091      * @return number of values of an {@link Sample} object
092      */
093     public int sampleSize() {
094         return _samples.get(0).size();
095     }
096 
097     /**
098      * Return the current {@link Sample} object. A newly created object is
099      * returned on demand.
100      *
101      * @return the current {@link Sample} object
102      */
103     public Sample currentSample() {
104         Sample sample = _samples.get(_samples.size() - 1);
105         if (sample.isFull()) {
106             sample = sample.newSample();
107             _samples.add(sample);
108         }
109 
110         return sample;
111     }
112 
113     /**
114      * Return the index of the next parameter index to calculate.
115      *
116      * @return the index of the next parameter index to calculate
117      */
118     public int nextParamIndex() {
119         return currentSample().nextIndex();
120     }
121 
122     /**
123      * Calculate the sample summary of this data object.
124      *
125      * @return the sample summary of this data object
126      */
127     public SampleSummary summary() {
128         return _samples.stream()
129             .filter(Sample::isFull)
130             .collect(toSampleSummary(sampleSize()));
131     }
132 
133     @Override
134     public int hashCode() {
135         return (37*_name.hashCode() + 17)*_samples.hashCode()*37 + 17;
136     }
137 
138     @Override
139     public boolean equals(final Object obj) {
140         return obj instanceof Data &&
141             _name.equals(((Data)obj)._name) &&
142             _samples.equals(((Data)obj)._samples);
143     }
144 
145     @Override
146     public String toString() {
147         return format("Data[name=%s, size=%d]", _name, dataSize());
148     }
149 
150     /**
151      * Create a new {@code Data} object with the given parameters.
152      *
153      * @param name the name of the data object
154      * @param samples the sample list of the data object
155      * @throws NullPointerException if one of the parameters is {@code null}
156      * @return a new {@code Data} object with the given parameters
157      */
158     public static Data of(final String name, final List<Sample> samples) {
159         return new Data(name, samples);
160     }
161 
162     /**
163      * Return a new {@code Data} object with the given name and the given number
164      * of parameters.
165      *
166      * @param name the name of the data object
167      * @param parameterCount the parameter count of the created samples
168      * @throws NullPointerException if the data {@code name} is {@code null}
169      * @throws IllegalArgumentException if the given {@code parameterCount} is
170      *         smaller then one
171      * @return a new {@code Data} object with the given parameters
172      */
173     public static Data of(final String name, final int parameterCount) {
174         return of(name, singletonList(Sample.of(parameterCount)));
175     }

178     /* *************************************************************************
179      *  XML reader/writer
180      * ************************************************************************/
181 
182     public static final Writer<Data> WRITER = elem(
183         "data",
184         attr("name").map(Data::getName),
185         elems(Sample.WRITER).map(d -> d._samples)
186     );
187 
188     @SuppressWarnings("unchecked")
189     public static final Reader<Data> READER = Reader.elem(
190         (Object[] v) -> Data.of((String)v[0], (List<Sample>)v[1]),
191         "data",
192         Reader.attr("name"),
193         Reader.elems(Sample.READER)
194     );
