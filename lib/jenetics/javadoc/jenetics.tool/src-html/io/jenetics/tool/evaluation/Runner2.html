020 package io.jenetics.tool.evaluation;
021 
022 import static java.util.Objects.requireNonNull;
023 
024 import java.nio.file.Path;
025 import java.nio.file.Paths;
026 import java.util.function.Function;
027 import java.util.function.Predicate;
028 import java.util.function.Supplier;
029 
030 import io.jenetics.Gene;
031 import io.jenetics.engine.Engine;
032 import io.jenetics.engine.EvolutionResult;
033 import io.jenetics.internal.util.Args;
034 import io.jenetics.tool.trial.TrialMeter;
035 import io.jenetics.xml.stream.Reader;
036 import io.jenetics.xml.stream.Writer;
037 
038 /**
039  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
040  * @version 4.0
041  * @since 3.5
042  */
043 public class Runner2<
044     P,
045     G extends Gene<?, G>,
046     N extends Number &  Comparable<? super N>
047 >
048     extends AbstractRunner<P>
049 {
050 
051     private final Engine<G, N> _engine1;
052     private final Function<? super P, Predicate<? super EvolutionResult<G, N>>> _terminator1;
053     private final Engine<G, N> _engine2;
054     private final Function<? super P, Predicate<? super EvolutionResult<G, N>>> _terminator2;
055 
056     public Runner2(
057         final Engine<G, N> engine1,
058         final Function<? super P, Predicate<? super EvolutionResult<G, N>>> terminator1,
059         final Engine<G, N> engine2,
060         final Function<? super P, Predicate<? super EvolutionResult<G, N>>> terminator2,
061         final Supplier<TrialMeter<P>> trialMeter,
062         final Writer<P> writer,
063         final Reader<P> reader,
064         final int sampleCount,
065         final Path resultPath
066     ) {
067         super(trialMeter, writer, reader, sampleCount, resultPath);
068         _engine1 = requireNonNull(engine1);
069         _terminator1 = requireNonNull(terminator1);
070         _engine2 = requireNonNull(engine2);
071         _terminator2 = requireNonNull(terminator2);
072     }
073 
074     protected double[] fitness(final P param) {
075         final Predicate<? super EvolutionResult<G, N>> terminator1 =
076             _terminator1.apply(param);
077 
078         final long start1 = System.currentTimeMillis();
079         final EvolutionResult<G, N> result1 = _engine1.stream()
080             .limit(terminator1)
081             .collect(EvolutionResult.toBestEvolutionResult());
082         final long end1 = System.currentTimeMillis();
083 
084         final Predicate<? super EvolutionResult<G, N>> terminator2 =
085             _terminator2.apply(param);
086 
087         final long start2 = System.currentTimeMillis();
088         final EvolutionResult<G, N> result2 = _engine2.stream()
089             .limit(terminator2)
090             .collect(EvolutionResult.toBestEvolutionResult());
091         final long end2 = System.currentTimeMillis();
092 
093         return new double[] {
094             result1.getTotalGenerations(),
095             result1.getBestFitness() != null
096                 ? result1.getBestFitness().doubleValue()
097                 : Double.NEGATIVE_INFINITY,
098             end1 - start1,
099 
100             result2.getTotalGenerations(),
101             result2.getBestFitness() != null
102                 ? result2.getBestFitness().doubleValue()
103                 : Double.NEGATIVE_INFINITY,
104             end2 - start2
105         };
106     }
107 
108     public static <P, G extends Gene<?, G>, N extends Number &  Comparable<? super N>>
109     Runner2<P, G, N> of(
110         final Engine<G, N> engine1,
111         final Function<? super P, Predicate<? super EvolutionResult<G, N>>> terminator1,
112         final Engine<G, N> engine2,
113         final Function<? super P, Predicate<? super EvolutionResult<G, N>>> terminator2,
114         final Supplier<TrialMeter<P>> trialMeter,
115         final Writer<P> writer,
116         final Reader<P> reader,
117         final String[] arguments
118     ) {
119         final Args args = Args.of(arguments);
120 
121         return new Runner2<>(
122             engine1,
123             terminator1,
124             engine2,
125             terminator2,
126             trialMeter,
127             writer,
128             reader,
129             args.intArg("sample-count")
130                 .orElse(50),
131             args.arg("result-file")
132                 .map(f -> Paths.get(f))
133                 .orElse(Paths.get("trial_meter.xml"))
134         );
135     }
