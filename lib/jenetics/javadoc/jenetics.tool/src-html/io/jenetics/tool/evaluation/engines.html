20 package io.jenetics.tool.evaluation;
21 
22 import java.util.Random;
23 
24 import io.jenetics.BitGene;
25 import io.jenetics.Mutator;
26 import io.jenetics.RouletteWheelSelector;
27 import io.jenetics.SinglePointCrossover;
28 import io.jenetics.TournamentSelector;
29 import io.jenetics.engine.Engine;
30 import io.jenetics.internal.util.require;
31 import io.jenetics.prngine.LCG64ShiftRandom;
32 import io.jenetics.tool.problem.Knapsack;
33 
34 /**
35  * Definition of commonly used testing {@link Engine} objects.
36  *
37  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
38  * @version 3.4
39  * @since 3.4
40  */
41 public final class engines {
42     private engines() {require.noInstance();}
43 
44     /**
45      * The test {@link Knapsack} {@link Engine} used for the evolution
46      * performance tests.
47      */
48     public static Engine<BitGene, Double>
49         KNAPSACK = knapsack(new LCG64ShiftRandom(10101));
50 
51     public static Engine<BitGene, Double> KNAPSACK(final int populationSize) {
52         return knapsack(populationSize, new LCG64ShiftRandom(10101));
53     }
54 
55     public static Engine<BitGene, Double> knapsack(final Random random) {
56         return knapsack(150, random);
57     }
58 
59     /**
60      * Create a new {@link Engine} for solving the {@link Knapsack} problem. The
61      * engine is used for testing purpose.
62      *
63      * @see Knapsack#of(int, Random)
64      *
65      * @param populationSize the population size of the created engine
66      * @param random the random engine used for creating the {@link Knapsack}
67      *        problem instance
68      * @return a new {@link Knapsack} solving evolution {@link Engine}
69      */
70     public static Engine<BitGene, Double> knapsack(
71         final int populationSize,
72         final Random random
73     ) {
74         // Search space fo 2^250 ~ 10^75.
75         final Knapsack knapsack = Knapsack.of(250, random);
76 
77         // Configure and build the evolution engine.
78         return Engine.builder(knapsack)
79             .populationSize(populationSize)
80             .survivorsSelector(new TournamentSelector<>(5))
81             .offspringSelector(new RouletteWheelSelector<>())
82             .alterers(
83                 new Mutator<>(0.03),
84                 new SinglePointCrossover<>(0.125))
85             .build();
86     }
87
