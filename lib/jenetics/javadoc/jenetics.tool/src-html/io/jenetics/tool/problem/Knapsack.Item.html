020package io.jenetics.tool.problem;
021
022import static java.lang.String.format;
023import static java.util.Objects.requireNonNull;
024
025import java.io.Serializable;
026import java.util.Random;
027import java.util.Set;
028import java.util.function.Function;
029import java.util.stream.Collector;
030import java.util.stream.Stream;
031
032import io.jenetics.BitGene;
033import io.jenetics.engine.Codec;
034import io.jenetics.engine.Codecs;
035import io.jenetics.engine.Problem;
036import io.jenetics.internal.util.require;
037import io.jenetics.tool.problem.Knapsack.Item;
038import io.jenetics.util.ISeq;
039
040/**
041 * <i>Canonical</i> definition of the <i>Knapsack</i> problem. This
042 * <i>reference</i> implementation is used for (evolution) performance tests of
043 * the GA {@link io.jenetics.engine.Engine}.
044 *
045 * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
046 * @version 3.4
047 * @since 3.4
048 */
049public final class Knapsack implements Problem<ISeq<Item>, BitGene, Double> {
050
051        /**
052         * This class represents a knapsack item with the specific <i>size</i> and
053         * <i>value</i>.
054         */
055        public static final class Item implements Serializable {
056                private static final long serialVersionUID = 1L;
057
058                private final double _size;
059                private final double _value;
060
061                private Item(final double size, final double value) {
062                        _size = require.nonNegative(size);
063                        _value = require.nonNegative(value);
064                }
065
066                /**
067                 * Return the item size.
068                 *
069                 * @return the item size
070                 */
071                public double getSize() {
072                        return _size;
073                }
074
075                /**
076                 * Return the item value.
077                 *
078                 * @return the item value
079                 */
080                public double getValue() {
081                        return _value;
082                }
083
084                @Override
085                public int hashCode() {
086                        int hash = 1;
087                        long bits = Double.doubleToLongBits(_size);
088                        hash = 31*hash + (int)(bits^(bits >>> 32));
089
090                        bits = Double.doubleToLongBits(_value);
091                        return 31*hash + (int)(bits^(bits >>> 32));
092                }
093
094                @Override
095                public boolean equals(final Object obj) {
096                        return obj instanceof Item &&
097                                Double.compare(_size, ((Item)obj)._size) == 0 &&
098                                Double.compare(_value, ((Item)obj)._value) == 0;
099                }
100
101                @Override
102                public String toString() {
103                        return format("Item[size=%f, value=%f]", _size, _value);
104                }
105
106                /**
107                 * Return a new knapsack {@code Item} with the given {@code size} and
108                 * {@code value}.
109                 *
110                 * @param size the item size
111                 * @param value the item value
112                 * @return a new knapsack {@code Item}
113                 * @throws IllegalArgumentException if one of the parameters is smaller
114                 *         then zero
115                 */
116                public static Item of(final double size, final double value) {
117                        return new Item(size, value);
118                }
119
120                /**
121                 * Create a new <i>random</i> knapsack item for testing purpose.
122                 *
123                 * @param random the random engine used for creating the knapsack item
124                 * @return a new <i>random</i> knapsack item
125                 * @throws NullPointerException if the random engine is {@code null}
126                 */
127                public static Item random(final Random random) {
128                        return new Item(random.nextDouble()*100, random.nextDouble()*100);
129                }
130
131                /**
132                 * Return a {@link Collector}, which sums the size and value of knapsack
133                 * items.
134                 *
135                 * @return a knapsack item sum {@link Collector}
136                 */
137                public static Collector<Item, ?, Item> toSum() {
138                        return Collector.of(
139                                () -> new double[2],
140                                (a, b) -> {a[0] += b._size; a[1] += b._value;},
141                                (a, b) -> {a[0] += b[0]; a[1] += b[1]; return a;},
142                                r -> new Item(r[0], r[1])
143                        );
144                }
145        }
146
147
148        private final Codec<ISeq<Item>, BitGene> _codec;
149        private final double _knapsackSize;
150
151        /**
152         * Create a new {@code Knapsack} definition with the given
153         *
154         * @param items the basic {@link Set} of knapsack items.
155         * @param knapsackSize the maximal knapsack size
156         * @throws NullPointerException if the {@code items} set is {@code null}
157         */
158        public Knapsack(final ISeq<Item> items, final double knapsackSize) {
159                _codec = Codecs.ofSubSet(items);
160                _knapsackSize = knapsackSize;
161        }
162
163        @Override
164        public Function<ISeq<Item>, Double> fitness() {
165                return items -> {
166                        final Item sum = items.stream().collect(Item.toSum());
167                        return sum._size <= _knapsackSize ? sum._value : 0;
168                };
169        }
170
171        @Override
172        public Codec<ISeq<Item>, BitGene> codec() {
173                return _codec;
174        }
175
176        /**
177         * Factory method for creating <i>same</i> Knapsack problems for testing
178         * purpose.
179         *
180         * @param itemCount the number of knapsack items in the basic set
181         * @param random the random engine used for creating random knapsack items.
182         *        This allows to create reproducible item sets and reproducible
183         *        {@code Knapsack} problems, respectively.
184         * @return a {@code Knapsack} problem object (for testing purpose).
185         */
186        public static Knapsack of(final int itemCount, final Random random) {
187                requireNonNull(random);
188
189                return new Knapsack(
190                        Stream.generate(() -> Item.random(random))
191                                .limit(itemCount)
192                                .collect(ISeq.toISeq()),
193                        itemCount*100.0/3.0
194                );
195        }
196
197}
