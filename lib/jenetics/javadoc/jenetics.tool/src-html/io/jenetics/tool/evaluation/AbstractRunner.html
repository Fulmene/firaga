020 package io.jenetics.tool.evaluation;
021 
022 import static java.util.Objects.requireNonNull;
023 
024 import java.io.Console;
025 import java.nio.file.Path;
026 import java.util.concurrent.atomic.AtomicBoolean;
027 import java.util.function.Supplier;
028 
029 import io.jenetics.internal.util.require;
030 import io.jenetics.tool.trial.Trial;
031 import io.jenetics.tool.trial.TrialMeter;
032 import io.jenetics.xml.stream.Reader;
033 import io.jenetics.xml.stream.Writer;
034 
035 /**
036  * @author <a href="mailto:franz.wilhelmstoetter@gmail.com">Franz Wilhelmstötter</a>
037  * @version 4.0
038  * @since 3.5
039  */
040 public abstract class AbstractRunner<P> {
041 
042     private final Supplier<TrialMeter<P>> _trialMeter;
043     private final Writer<P> _writer;
044     private final Reader<P> _reader;
045     private final int _sampleCount;
046     private final Path _resultPath;
047 
048     private volatile Thread _trialThread = null;
049     private final AtomicBoolean _stop = new AtomicBoolean(false);
050 
051     protected AbstractRunner(
052         final Supplier<TrialMeter<P>> trialMeter,
053         final Writer<P> writer,
054         final Reader<P> reader,
055         final int sampleCount,
056         final Path resultPath
057     ) {
058         _trialMeter = requireNonNull(trialMeter);
059         _writer = requireNonNull(writer);
060         _reader = requireNonNull(reader);
061         _sampleCount = require.positive(sampleCount);
062         _resultPath = requireNonNull(resultPath);
063     }
064 
065     protected abstract double[] fitness(final P param);
066 
067     public void start() {
068         if (_trialThread != null) {
069             throw new IllegalStateException("Trial thread already running.");
070         }
071 
072         final Trial<P> trial = new Trial<>(
073             this::fitness,
074             _trialMeter,
075             _writer,
076             _reader,
077             count -> count >= _sampleCount || _stop.get(),
078             _resultPath
079         );
080 
081         _trialThread = new Thread(trial);
082         _trialThread.start();
083     }
084 
085     public void join() throws InterruptedException {
086         if (_trialThread == null) {
087             throw new IllegalStateException("Trial thread is not running.");
088         }
089 
090         try {
091             final Console console = System.console();
092             if (console != null) {
093                 final Thread interrupter = new Thread(() -> {
094                     String command;
095                     do {
096                         command = console.readLine();
097                         Trial.info("Got command '" + command + "'");
098                     } while (!"exit".equals(command));
099 
100                     Trial.info("Stopping trial...");
101                     _trialThread.interrupt();
102                 });
103                 interrupter.setName("Console read thread");
104                 interrupter.setDaemon(true);
105                 interrupter.start();
106             }
107 
108             _trialThread.join();
109             Trial.info("Sopped trial.");
110         } finally {
111             _trialThread = null;
112         }
113     }
114 
115 }
