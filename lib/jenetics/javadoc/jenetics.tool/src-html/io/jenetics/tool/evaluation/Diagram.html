131     /**
132      * Create a performance diagram.
133      *
134      * @param input the input data
135      * @param template the Gnuplot template to use
136      * @param params the diagram parameters (x-axis)
137      * @param output the output file
138      * @param summary the first summary data
139      * @param summaries the rest of the summary data
140      * @throws IOException if the diagram generation fails
141      * @throws NullPointerException of one of the parameters is {@code null}
142      * @throws IllegalArgumentException if the {@code params}, {@code generation}
143      *         and {@code fitness} doesn't have the same parameter count
144      */
145     public static void create(
146         final Path input,
147         final Template template,
148         final Params<?> params,
149         final Path output,
150         final SampleSummary summary,
151         final SampleSummary... summaries
152     )
153         throws IOException
154     {
155         final Stream<SampleSummary> summaryStream = Stream.concat(
156             Stream.of(summary), Stream.of(summaries)
157         );
158         summaryStream.forEach(s -> {
159             if (params.size() != s.parameterCount()) {
160                 throw new IllegalArgumentException(format(
161                     "Parameters have different size: %d", params.size()
162                 ));
163             }
164         });
165 
166         final Path templatePath = tempPath();
167         try {
168             IO.write(template.content(), templatePath);
169 
170             final Path dataPath = tempPath();
171             try {
172                 final String data = IntStream.range(0, params.size())
173                     .mapToObj(i -> toLineString(i, params, summary, summaries))
174                     .collect(Collectors.joining("\n"));
175                 IO.write(data, dataPath);
176 
177                 final Gnuplot gnuplot = new Gnuplot(templatePath);
178                 gnuplot.setEnv(params(input));
179                 gnuplot.create(dataPath, output);
180             } finally {
181                 deleteIfExists(dataPath);
182             }
183         } finally {
184             deleteIfExists(templatePath);
185         }
186     }
187 
188     private static Path tempPath() throws IOException {
189         return createTempFile("__diagram_template__", "__").toPath();
190     }
191 
192     private static String toLineString(
193         final int index,
194         final Params<?> params,
195         final SampleSummary summary,
196         final SampleSummary... summaries
197     ) {
198         return concat(concat(
199                 Stream.of(params.get(index).toString().split(":")),
200                 DoubleStream.of(summary.getPoints().get(index).toArray())
201                     .mapToObj(Double::toString)),
202                 Stream.of(summaries)
203                     .flatMapToDouble(s -> DoubleStream.of(s.getPoints().get(index).toArray()))
204                     .mapToObj(Double::toString))
205             .collect(Collectors.joining(" "));
206     }

208     public static void main(final String[] arguments) throws Exception {
209         final Args args = Args.of(arguments);
210 
211         final Path input = args.arg("input")
212             .map(Paths::get)
213             .map(Path::toAbsolutePath)
214             .get();
215 
216         final String[] samples = args.arg("samples")
217             .map(s -> s.split(","))
218             .orElse(new String[]{"Generation", "Fitness"});
219 
220         final TrialMeter<String> trial = TrialMeter.read(
221             input,
222             Reader.text()
223         );
224         final Params<String> params = trial.getParams();
225         final SampleSummary summary = trial.getData(samples[0]).summary();
226         final SampleSummary[] summaries = Arrays.stream(samples, 1, samples.length)
227             .map(s -> trial.getData(s).summary())
228             .toArray(SampleSummary[]::new);
229 
230         create(
231             input,
232             template(input),
233             params,
234             output(input),
235             summary,
236             summaries
237         );
238     }

240     private static Template template(final Path path) {
241         final String name = path.getFileName().toString()
242             .split("-")[1]
243             .split("\\.")[0];
244 
245         return Arrays.stream(Template.values())
246             .filter(t -> t.getName().equals(name))
247             .findFirst().get();
248     }
249 
250     private static Map<String, String> params(final Path path) {
251         System.out.println(path.getFileName());
252         final List<String> parts = param(path.getFileName().toString())
253             .flatMap(p -> Stream.of(p.split("@")))
254             .collect(Collectors.toList());
255 
256         final Map<String, String> params = new HashMap<>();
257         for (int i = 0; i < parts.size(); ++i) {
258             final String key = format("PARAM_%s", i);
259             params.put(key, parts.get(i));
260         }
261 
262         return params;
263     }
264 
265     private static Stream<String> param(final String name) {
266         final String[] parts = name.split("-");
267         return parts.length == 3
268             ? Stream.of(parts[2].split("\\.")[0])
269             : Stream.empty();
270     }
271 
272     private static Path output(final Path path) {
273         final String name = path.getFileName().toString().split("\\.")[0];
274         return Paths.get(path.getParent().toString(), name + ".svg");
275     }
